    25	func ParseDN(dn string) (*pkix.Name, error) {
    26	    name := &pkix.Name{}
    27	    
    28	    // Try to parse both formats: slash-notation and comma-separated
    29	    // This is a simplified parser - in production, use a proper DN parser
    30	    if len(dn) == 0 {
    31	        return nil, fmt.Errorf("empty DN string")
    32	    }
    33	    
    34	    // Basic parsing for demo purposes
    35	    // Format: "/CN=Common Name/O=Organization/C=US" or "CN=Common Name,O=Organization"
    36	    current := ""
    37	    for i := 0; i < len(dn); i++ {
    38	        if dn[i] == '/' && i == 0 {
    39	            continue
    40	        }
    41	        if dn[i] == '/' || dn[i] == ',' {
    42	            if current != "" {
    43	                parts := splitRDN(current)
    44	                if len(parts) == 2 {
    45	                    switch parts[0] {
    46	                    case "CN":
    47	                        name.CommonName = parts[1]
    48	                    case "O":
    49	                        name.Organization = []string{parts[1]}
    50	                    case "OU":
    51	                        name.OrganizationalUnit = []string{parts[1]}
    52	                    case "C":
    53	                        name.Country = []string{parts[1]}
    54	                    case "ST":
    55	                        name.Province = []string{parts[1]}
    56	                    case "L":
    57	                        name.Locality = []string{parts[1]}
    58	                    }
    59	                }
    60	            }
    61	            current = ""
    62	        } else {
    63	            current += string(dn[i])
    64	        }
    65	    }
    66	    
    67	    if current != "" {
    68	        parts := splitRDN(current)
    69	        if len(parts) == 2 {
    70	            switch parts[0] {
    71	            case "CN":
    72	                name.CommonName = parts[1]
    73	            case "O":
    74	                name.Organization = []string{parts[1]}
    75	            case "OU":
    76	                name.OrganizationalUnit = []string{parts[1]}
    77	            case "C":
    78	                name.Country = []string{parts[1]}
    79	            case "ST":
    80	                name.Province = []string{parts[1]}
    81	            case "L":
    82	                name.Locality = []string{parts[1]}
    83	            }
    84	        }
    85	    }
    86	    
    87	    if name.CommonName == "" {
    88	        return nil, fmt.Errorf("DN must contain a Common Name (CN)")
    89	    }
    90	    
    91	    return name, nil
    92	}

   111	func CreateSelfSignedRootCA(config *CertificateConfig) (*x509.Certificate, error) {
   112	    name, err := ParseDN(config.Subject)
   113	    if err != nil {
   114	        return nil, err
   115	    }
   116	    
   117	    notBefore := time.Now().UTC()
   118	    notAfter := notBefore.AddDate(0, 0, config.ValidityDays)
   119	    
   120	    // Compute Subject Key Identifier (SHA-1 of public key)
   121	    var pubKeyBytes []byte
   122	    switch pub := config.PublicKey.(type) {
   123	    case *rsa.PublicKey:
   124	        pubKeyBytes, err = x509.MarshalPKIXPublicKey(pub)
   125	    case *ecdsa.PublicKey:
   126	        pubKeyBytes, err = x509.MarshalPKIXPublicKey(pub)
   127	    default:
   128	        return nil, fmt.Errorf("unsupported public key type")
   129	    }
   130	    if err != nil {
   131	        return nil, fmt.Errorf("failed to marshal public key: %w", err)
   132	    }
   133	    
   134	    hash := sha1.Sum(pubKeyBytes)
   135	    ski := hash[:]
   136	    
   137	    template := &x509.Certificate{
   138	        SerialNumber: config.SerialNumber,
   139	        Subject:      *name,
   140	        Issuer:       *name, // Self-signed, so issuer = subject
   141	        NotBefore:    notBefore,
   142	        NotAfter:     notAfter,
   143	        
   144	        KeyUsage: x509.KeyUsageCertSign | x509.KeyUsageCRLSign,
   145	        
   146	        BasicConstraintsValid: true,
   147	        IsCA:                  true,
   148	        MaxPathLen:            0, // No path length constraint
   149	        
   150	        SubjectKeyId:   ski,
   151	        AuthorityKeyId: ski, // Same as SKI for self-signed
   152	        
   153	        SignatureAlgorithm: getSignatureAlgorithm(config.KeyType),
   154	    }
   155	    
   156	    return template, nil
   157	}

   186	func VerifyCertificate(certPath string) error {
   187	    certPEM, err := os.ReadFile(certPath)
   188	    if err != nil {
   189	        return fmt.Errorf("failed to read certificate: %w", err)
   190	    }
   191	    
   192	    block, _ := pem.Decode(certPEM)
   193	    if block == nil {
   194	        return fmt.Errorf("failed to decode PEM certificate")
   195	    }
   196	    
   197	    cert, err := x509.ParseCertificate(block.Bytes)
   198	    if err != nil {
   199	        return fmt.Errorf("failed to parse certificate: %w", err)
   200	    }
   201	    
   202	    // Verify self-signed certificate
   203	    roots := x509.NewCertPool()
   204	    roots.AddCert(cert)
   205	    
   206	    opts := x509.VerifyOptions{
   207	        Roots: roots,
   208	    }
   209	    
   210	    if _, err := cert.Verify(opts); err != nil {
   211	        return fmt.Errorf("certificate verification failed: %w", err)
   212	    }

   214	    return nil

    16	
    17	const (
    18	    RSA KeyType = "rsa"
    19	    ECC KeyType = "ecc"
    20	)
    21	
    22	type KeyPair struct {
    23	    PrivateKey interface{}
    24	    PublicKey  interface{}
    25	    KeyType    KeyType
    26	    KeySize    int
    27	}
    28	
    29	func GenerateKeyPair(keyType KeyType, keySize int) (*KeyPair, error) {
    30	    switch keyType {
    31	    case RSA:
    32	        if keySize != 4096 {
    33	            return nil, fmt.Errorf("RSA key size must be 4096 bits")
    34	        }
    35	        privateKey, err := rsa.GenerateKey(rand.Reader, keySize)
    36	        if err != nil {
    37	            return nil, fmt.Errorf("failed to generate RSA key: %w", err)
    38	        }
    39	        return &KeyPair{
    40	            PrivateKey: privateKey,
    41	            PublicKey:  &privateKey.PublicKey,
    42	            KeyType:    RSA,
    43	            KeySize:    keySize,
    44	        }, nil
    45	        
    46	    case ECC:
    47	        if keySize != 384 {
    48	            return nil, fmt.Errorf("ECC key size must be 384 bits (P-384)")
    49	        }
    50	        privateKey, err := ecdsa.GenerateKey(elliptic.P384(), rand.Reader)
    51	        if err != nil {
    52	            return nil, fmt.Errorf("failed to generate ECC key: %w", err)
    53	        }
    54	        return &KeyPair{
    55	            PrivateKey: privateKey,
    56	            PublicKey:  &privateKey.PublicKey,
    57	            KeyType:    ECC,
    58	            KeySize:    keySize,
    59	        }, nil
    60	        
    61	    default:
    62	        return nil, fmt.Errorf("unsupported key type: %s", keyType)
    63	    }
    64	}
    65	
    66	func EncryptPrivateKey(privateKey interface{}, passphrase []byte) ([]byte, error) {
    67	    var keyBytes []byte
    68	    var err error
    69	    
    70	    switch k := privateKey.(type) {
    71	    case *rsa.PrivateKey:
    72	        keyBytes = x509.MarshalPKCS1PrivateKey(k)
    73	    case *ecdsa.PrivateKey:
    74	        keyBytes, err = x509.MarshalECPrivateKey(k)
    75	        if err != nil {
    76	            return nil, fmt.Errorf("failed to marshal ECC private key: %w", err)
    77	        }
    78	    default:
    79	        return nil, fmt.Errorf("unsupported private key type")
    80	    }
    81	    
    82	    // Create encrypted PEM block
    83	    block, err := x509.EncryptPEMBlock(rand.Reader, "PRIVATE KEY", keyBytes, passphrase, x509.PEMCipherAES256)
    84	    if err != nil {
    85	        return nil, fmt.Errorf("failed to encrypt private key: %w", err)
    86	    }
    87	    
    88	    return pem.EncodeToMemory(block), nil
    89	}
    90	
    91	func SaveEncryptedKey(keyPair *KeyPair, passphrase []byte, path string, perm os.FileMode) error {
    92	    encryptedKey, err := EncryptPrivateKey(keyPair.PrivateKey, passphrase)
    93	    if err != nil {
    94	        return err
    95	    }
    96	    
    97	    return ioutil.WriteFile(path, encryptedKey, perm)
    98	}
